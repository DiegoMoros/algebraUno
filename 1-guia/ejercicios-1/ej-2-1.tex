	\begin{enumerate}[label=(\roman*)]

		\item $\set{3} \en A    $

		\item $\vacio \en A  $
		\item $\vacio \subseteq A  $
		\item $A \en A  $
		\item $A \subseteq A  $
	\end{enumerate}
\end{multicols}
