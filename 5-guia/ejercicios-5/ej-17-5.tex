\begin{enunciado}{\ejercicio}
  Probar que para todo $a\en \enteros$ vale
  \begin{multicols}{2}
    \begin{enumerate}[label=\roman*)]
      \item $728 \divideA a^{27} - a^3$
      \item $\frac{2a^7}{35} + \frac{a}{7} - \frac{a^3}{5}  \en \enteros$
    \end{enumerate}
  \end{multicols}
\end{enunciado}
