          \begin{itemize}

            \item $f(1) = -12$,
          \end{itemize}

    \item Con el polinomio $f$ hallado exprasar factorización en irreducibles en $\racionales[X], \reales[X] \ytext \complejos[X]$.
  \end{enumerate}
